Air-Gapped Prerequisites

Mirror packages into an internal repository or copy release packages plus checksums.txt to each node. The installation commands stay the same after the internal repository is configured.

: "${LI9_RPM_REPO_URL:?set internal RPM repository URL}"
: "${LI9_DEB_REPO_URL:?set internal DEB repository URL}"
: "${LI9_DEB_GPG_URL:?set internal DEB GPG key URL}"

# RPM repository
sudo tee /etc/yum.repos.d/li9-s3.repo >/dev/null <<EOF
[li9-s3]
name=Li9 S3 Storage
baseurl=${LI9_RPM_REPO_URL}
enabled=1
gpgcheck=1
gpgkey=${LI9_RPM_REPO_URL}/RPM-GPG-KEY-li9
EOF

# DEB repository
curl -fsSL "${LI9_DEB_GPG_URL}" | sudo gpg --dearmor -o /usr/share/keyrings/li9-s3.gpg
echo "deb [signed-by=/usr/share/keyrings/li9-s3.gpg] ${LI9_DEB_REPO_URL} stable main" | sudo tee /etc/apt/sources.list.d/li9-s3.list
sudo apt-get update
